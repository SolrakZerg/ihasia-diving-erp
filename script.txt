// Nombre de la pestaña en tu Hoja de Cálculo donde se guardarán los datos
var SHEET_NAME = "NewFormulario"; 

function doPost(e) {
  Logger.log("doPost: EJECUCIÓN INICIADA."); // Log al inicio
  // Logger.log("doPost: Parámetros recibidos: " + JSON.stringify(e.parameters)); // Descomenta si necesitas verlos

  var lock = LockService.getScriptLock();
  var locked = false; // Para saber si necesitamos liberar el lock
  try {
    locked = lock.tryLock(10000); 
    if (!locked) {
      Logger.log("doPost: No se pudo obtener el lock del script.");
      // Devolver un error aquí es mejor que continuar sin lock si es crítico
      return ContentService
        .createTextOutput(JSON.stringify({"result":"error", "message": "Timeout al obtener lock. Intente de nuevo."}))
        .setMimeType(ContentService.MimeType.JSON);
    }
    Logger.log("doPost: Lock obtenido.");

    var doc = SpreadsheetApp.getActiveSpreadsheet();
    var sheet = doc.getSheetByName(SHEET_NAME);

    if (!sheet) {
      Logger.log("doPost: ERROR - Hoja '" + SHEET_NAME + "' no encontrada.");
      // Es importante devolver un error para que PHP sepa que algo falló
      return ContentService
        .createTextOutput(JSON.stringify({"result":"error", "message": "Hoja de destino '" + SHEET_NAME + "' no encontrada."}))
        .setMimeType(ContentService.MimeType.JSON);
    }
    // ... resto de tu lógica de try ...
    
    var headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0];
    var nextRow = sheet.getLastRow() + 1; 
    var newRowData = [];

    for (var i = 0; i < headers.length; i++) {
      var headerName = headers[i];
      var paramValue = e.parameter[headerName];
      var headerNameLower = headerName.toLowerCase();

      if (headerNameLower === 'timestamp' && !paramValue) {
        newRowData.push(new Date());
      } else if (headerName === 'WhatsApp') { 
        var whatsappNumber = paramValue ? paramValue : "";
        if (whatsappNumber) {        
          newRowData.push("wa.me/" + whatsappNumber.replace(/[^\d+]/g, ''));
        } else {
          newRowData.push("");
        }
      } else {
        newRowData.push(paramValue ? paramValue : ""); 
      }
    }
    
    if (newRowData.length > 0) {
        sheet.getRange(nextRow, 1, 1, newRowData.length).setValues([newRowData]);
        Logger.log("doPost: Nueva fila escrita en la hoja en la fila " + nextRow);
    } else {
        Logger.log("doPost: No se prepararon datos para la nueva fila.");
    }

    return ContentService
      .createTextOutput(JSON.stringify({"result":"success", "row": nextRow, "data_received": e.parameters}))
      .setMimeType(ContentService.MimeType.JSON);

  } catch(error) {
    Logger.log("doPost: ERROR CAPTURADO: " + error.toString() + " Stack: " + error.stack);
    // Logger.log("doPost: Parámetros que causaron el error: " + JSON.stringify(e.parameters)); // Descomenta si es útil
    return ContentService
      .createTextOutput(JSON.stringify({"result":"error", "message": error.message, "stack": error.stack, "parameters_received": e.parameters}))
      .setMimeType(ContentService.MimeType.JSON);
  } finally {
    if (locked) { 
      lock.releaseLock();
      Logger.log("doPost: Lock liberado.");
    }
    Logger.log("doPost: EJECUCIÓN FINALIZADA (dentro de finally)."); // <-- LOG DE FINALIZACIÓN AQUÍ
  }
  // Cualquier Logger.log() aquí fuera no se ejecutará
}
